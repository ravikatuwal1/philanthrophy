CREATE TABLE galleries (
  id VARCHAR(100) PRIMARY KEY,
  title VARCHAR(255),
  description TEXT,
  cover VARCHAR(255)
);

CREATE TABLE gallery_images (
  id INT AUTO_INCREMENT PRIMARY KEY,
  gallery_id VARCHAR(100),
  url VARCHAR(255),
  FOREIGN KEY (gallery_id) REFERENCES galleries(id)
);
